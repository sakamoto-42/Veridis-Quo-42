$(document).ready(function() {						// attend que le document soit prêt avant que le code à l'intérieur de ce bloc (des accolades) 											// soit exécuté (pour que la page soit manipulée sans risque)
	function randomColorChannel() {
		return Math.floor(Math.random() * 256);	
	}

	function randomAlphaChannel() {
		return Math.round(Math.random() * 100) / 100;
	}

	function randomRGBA() {
		let R = randomColorChannel();
		let G = randomColorChannel();
		let B = randomColorChannel();
		let A = randomAlphaChannel();
	
		return `rgba(${R}, ${G}, ${B}, ${A})`;
	}

	$("#random-bg-btn").on("click", function() {			// sélectionne l'élément d'id random-bg-btn et lui attribue l'event listener "on click", avec la 										// fonction anonyme suivante
		$("body").css("background-color", randomRGBA());	// sélectionne le body et lui applique pour la propriété css background-color la valeur retournée par 									 	// randomRGBA()
	});
});
