$(document).ready(function() {						// JQUERY : attend que le document soit prêt avant que le code à l'intérieur de ce bloc (des accolades) 									// soit exécuté (pour que la page soit manipulée sans risque)
	function randomColorChannel() {
		return Math.floor(Math.random() * 256);	
	}

	function randomRGB() {
		let R = randomColorChannel();
		let G = randomColorChannel();
		let B = randomColorChannel();
	
		return `rgb(${R}, ${G}, ${B})`;
	}

	$("#random-bg-btn").on("click", function() {			// JQUERY : sélectionne l'élément d'id random-bg-btn et lui attribue l'event listener "click"  											// la fonction anonyme suivante sera exécutée lors du clic

		$("body").css("background-color", randomRGB());		// JQUERY : sélectionne le body et lui applique la propriété CSS background-color avec comme valeur  									 	// celle retournée par randomRGB()
	});
});
