<!DOCTYPE html>
<html lang="en">		
	<head>
		<title>Balloon</title>
		<meta charset="UTF-8">
		<meta http-equiv="X-UA-Compatible" content="IE=edge">
		<meta name="viewport" content="width=device-width, initial-scale=1.0">
		<style>
			html, body, .container {
				height: 100%;
			}

			body {
				margin: 0;				
			}

			.container {
				display: flex;
				justify-content: center;
				align-items: center;
			}

			#balloon {				
				height: 200px;
				width: 200px;
				background-color: red;
				border-radius: 50%;
				cursor: pointer;
		</style>
	</head>
	<body>
		<div class="container">
			<div id="balloon"><div>
		</div>
		<script src="jquery-3.7.1.js"></script>						// lien vers jquery
		<script>
			$(document).ready(function() {
				const balloon = $("#balloon");					// JQUERY : sélectionne l'élément d'id balloon
				const balloonBackgroundColors = ['red', 'green', 'blue'];
				const growValue = 10;
				const shrinkValue = 5;
				const balloonMinSize = 200;
				const balloonMaxSize = 420;				

				let balloonSizeValue = balloonMinSize;
				let i = 0;

				balloon.on("click mouseleave", function(event) {		// JQUERY : ajoute les event listeners "click" et "mouseleave"
												// la fonction anonyme suivante sera exécutée lors du clic
												// avec comme paramètre l'évènement qui l'a déclenchée
						changeSize(event.type);				// on exécute la fonction changeSize avec le type d'évènement en paramètre
						changeColor(event.type);			// idem pour changeColor
				});

				function changeSize(action) {					// le paramètre est récupéré dans "action"
					if (action === "click")
						balloonSizeValue = ((balloonSizeValue + growValue) > balloonMaxSize) ? balloonMinSize : balloonSizeValue + growValue ;
					else if (action === "mouseleave")			
						balloonSizeValue = ((balloonSizeValue - shrinkValue) < balloonMinSize) ? balloonMinSize : balloonSizeValue - shrinkValue;
					
					balloon.height(balloonSizeValue + "px");	// JQUERY : attribution de la nouvelle hauteur au ballon
					balloon.width(balloonSizeValue + "px");		// JQUERY : attribution de la nouvelle largeur au ballon
				}

				function changeColor(action) {
					if (action === "click")
						i = adjustIndex(i + 1, balloonBackgroundColors.length);
					else if (action === "mouseleave")
						i = adjustIndex(i - 1, balloonBackgroundColors.length);

					balloon.css("background-color", balloonBackgroundColors[i]);	// JQUERY : attribution de la nouvelle couleur au ballon
				}

				function adjustIndex(index, arrayLength) {
					return (index + arrayLength) % arrayLength;									 
				}
			});
		</script>
	</body>					
</html>
